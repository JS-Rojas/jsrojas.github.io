// 1. Simulación de Red Neuronal / Fase de Griffiths
const NetworkViz = () => {
  const canvasRef = useRef(null);
  
  useEffect(() => {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    let animationFrameId;
    
    // Configuración de nodos
    const nodes = Array.from({ length: 30 }, () => ({
      x: Math.random() * canvas.width,
      y: Math.random() * canvas.height,
      vx: (Math.random() - 0.5) * 1,
      vy: (Math.random() - 0.5) * 1,
      active: Math.random() > 0.8
    }));

    const render = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      
      // Actualizar y dibujar nodos
      nodes.forEach(node => {
        node.x += node.vx;
        node.y += node.vy;
        
        // Rebote en bordes
        if (node.x < 0 || node.x > canvas.width) node.vx *= -1;
        if (node.y < 0 || node.y > canvas.height) node.vy *= -1;
        
        // Cambio de estado aleatorio (simulando disparo neuronal)
        if (Math.random() > 0.98) node.active = !node.active;

        ctx.beginPath();
        ctx.arc(node.x, node.y, node.active ? 4 : 2, 0, Math.PI * 2);
        ctx.fillStyle = node.active ? '#ef4444' : '#cbd5e1';
        ctx.fill();
      });

      // Dibujar conexiones (sinapsis dinámicas)
      ctx.strokeStyle = 'rgba(100, 116, 139, 0.2)';
      for (let i = 0; i < nodes.length; i++) {
        for (let j = i + 1; j < nodes.length; j++) {
          const dx = nodes[i].x - nodes[j].x;
          const dy = nodes[i].y - nodes[j].y;
          const dist = Math.sqrt(dx * dx + dy * dy);
          
          if (dist < 60) {
            ctx.beginPath();
            ctx.moveTo(nodes[i].x, nodes[i].y);
            ctx.lineTo(nodes[j].x, nodes[j].y);
            // Si ambos activos, conexión fuerte (Hebbian-ish)
            if (nodes[i].active && nodes[j].active) {
                ctx.strokeStyle = 'rgba(239, 68, 68, 0.4)';
                ctx.lineWidth = 1.5;
            } else {
                ctx.strokeStyle = 'rgba(100, 116, 139, 0.1)';
                ctx.lineWidth = 0.5;
            }
            ctx.stroke();
          }
        }
      }
      animationFrameId = requestAnimationFrame(render);
    };
    render();
    return () => cancelAnimationFrame(animationFrameId);
  }, []);

  return <canvas ref={canvasRef} width={400} height={200} className="w-full h-full rounded bg-slate-50" />;
};

// 2. Simulación de Ondas Culturales (Senoide con ruido)
const OscillationsViz = () => {
  const canvasRef = useRef(null);
  
  useEffect(() => {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    let time = 0;
    let animationFrameId;

    const render = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.beginPath();
      ctx.moveTo(0, canvas.height / 2);

      for (let x = 0; x < canvas.width; x++) {
        // Combinación de ondas para simular complejidad
        const y = canvas.height / 2 + 
                 Math.sin(x * 0.05 + time) * 30 + 
                 Math.sin(x * 0.1 - time * 0.5) * 15;
        ctx.lineTo(x, y);
      }

      ctx.strokeStyle = '#0ea5e9';
      ctx.lineWidth = 2;
      ctx.stroke();
      
      // Dibujar "eventos" o picos
      time += 0.05;
      animationFrameId = requestAnimationFrame(render);
    };
    render();
    return () => cancelAnimationFrame(animationFrameId);
  }, []);

  return <canvas ref={canvasRef} width={400} height={200} className="w-full h-full rounded bg-slate-50" />;
};

            <div className="px-2 pt-2 pb-3 space-y-1 sm:px-3">
              <button onClick={() => {setActiveSection('home'); setIsMobileMenuOpen(false)}} className="block w-full text-left px-3 py-2 text-base font-medium text-slate-700 hover:bg-slate-50">Home</button>
              <button onClick={() => {setActiveSection('research'); setIsMobileMenuOpen(false)}} className="block w-full text-left px-3 py-2 text-base font-medium text-slate-700 hover:bg-slate-50">Research</button>
              <button onClick={() => {setActiveSection('events'); setIsMobileMenuOpen(false)}} className="block w-full text-left px-3 py-2 text-base font-medium text-slate-700 hover:bg-slate-50">Events</button>
              <button onClick={() => {setActiveSection('cv'); setIsMobileMenuOpen(false)}} className="block w-full text-left px-3 py-2 text-base font-medium text-slate-700 hover:bg-slate-50">CV</button>
            </div>
          </div>
        )}
      </nav>

      {/* Main Content */}
      <main className="max-w-5xl mx-auto px-4 sm:px-6 lg:px-8 pt-24 pb-16">
        {renderContent()}
      </main>

      {/* Footer */}
      <footer className="border-t border-slate-200 bg-white py-8 mt-12">
        <div className="max-w-5xl mx-auto px-4 text-center text-slate-500 text-sm">
          <p className="mb-2">&copy; {new Date().getFullYear()} Juan S. Rojas. Built with React (and Physics).</p>
          <div className="flex justify-center gap-4 mt-4">
             <a href="#" className="hover:text-slate-900 transition-colors">Twitter</a>
             <a href="#" className="hover:text-slate-900 transition-colors">Google Scholar</a>
             <a href="#" className="hover:text-slate-900 transition-colors">ResearchGate</a>
          </div>
        </div>
      </footer>
    </div>
  );
}
