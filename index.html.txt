        }

        // 3. Scroll Reveal Animation
        const observerOptions = {
            threshold: 0.1,
            rootMargin: "0px 0px -50px 0px"
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                }
            });
        }, observerOptions);

        document.querySelectorAll('.reveal').forEach(el => observer.observe(el));

        // 4. IMMERSIVE CANVAS VISUALIZATIONS

        // --- GLOBAL BACKGROUND: FLOW FIELDS ---
        function initBgCanvas() {
            const canvas = document.getElementById('bgCanvas');
            const ctx = canvas.getContext('2d');
            let width, height, particles = [];

            function resize() {
                width = window.innerWidth;
                height = window.innerHeight;
                canvas.width = width;
                canvas.height = height;
                initParticles();
            }

            function initParticles() {
                particles = [];
                // Density
                const count = Math.floor(width * height / 4000); 
                for(let i=0; i<count; i++) {
                    particles.push({
                        x: Math.random() * width,
                        y: Math.random() * height,
                        vx: 0,
                        vy: 0,
                        size: Math.random() * 1.5 + 0.5,
                        life: Math.random() * 100
                    });
                }
            }

            function animate() {
                // Fading trails effect
                ctx.fillStyle = 'rgba(2, 6, 23, 0.05)'; // Fade out trails slowly
                ctx.fillRect(0, 0, width, height);
                
                ctx.fillStyle = "rgba(56, 189, 248, 0.15)"; // Light Sky Blue

                for(let i=0; i<particles.length; i++) {
                    let p = particles[i];
                    
                    // FLOW FIELD LOGIC
                    const angle = (Math.cos(p.x * 0.005) + Math.sin(p.y * 0.005)) * Math.PI;
                    
                    p.vx = Math.cos(angle) * 0.5; // Speed
                    p.vy = Math.sin(angle) * 0.5;

                    p.x += p.vx;
                    p.y += p.vy;

                    // Wrap around screen
                    if(p.x < 0) p.x = width;
                    if(p.x > width) p.x = 0;
                    if(p.y < 0) p.y = height;
                    if(p.y > height) p.y = 0;

                    ctx.beginPath();
                    ctx.arc(p.x, p.y, p.size, 0, Math.PI*2);
                    ctx.fill();
                }
                requestAnimationFrame(animate);
            }

            window.addEventListener('resize', resize);
            resize();
            animate();
        }

        // --- RESEARCH VISUALIZATION 1: NEURAL GLOW ---
        function initNetworkViz() {
            const canvas = document.getElementById('networkCanvas');
            if(!canvas) return;
            const ctx = canvas.getContext('2d');
            
            function resize() {
                if(canvas.parentNode) {
                    canvas.width = canvas.parentNode.getBoundingClientRect().width;
                    canvas.height = canvas.parentNode.getBoundingClientRect().height;
                }
            }
            window.addEventListener('resize', resize);
            setTimeout(resize, 100); 

            const nodes = Array.from({length: 45}, () => ({
                x: Math.random() * 800, y: Math.random() * 400,
                vx: (Math.random()-0.5), vy: (Math.random()-0.5),
                active: false
            }));

            function animate() {
                if(!canvas.isConnected) return;
                ctx.clearRect(0, 0, canvas.width, canvas.height);

                nodes.forEach(node => {
                    node.x += node.vx; node.y += node.vy;
                    if(node.x < 0 || node.x > canvas.width) node.vx *= -1;
                    if(node.y < 0 || node.y > canvas.height) node.vy *= -1;
                    if(Math.random() > 0.98) node.active = !node.active; 

                    if(node.active) {
                        ctx.shadowBlur = 15;
                        ctx.shadowColor = "#ef4444"; 
                        ctx.fillStyle = "#fca5a5";
                    } else {
                        ctx.shadowBlur = 0;
                        ctx.fillStyle = "#475569"; 
                    }
                    
                    ctx.beginPath();
                    ctx.arc(node.x, node.y, node.active ? 3 : 2, 0, Math.PI*2);
                    ctx.fill();
                    ctx.shadowBlur = 0; 
                });

                ctx.lineWidth = 1;
                for(let i=0; i<nodes.length; i++) {
                    for(let j=i+1; j<nodes.length; j++) {
                        let dx = nodes[i].x - nodes[j].x;
                        let dy = nodes[i].y - nodes[j].y;
                        let dist = Math.sqrt(dx*dx+dy*dy);
                        if(dist < 70) {
                            ctx.beginPath();
                            ctx.moveTo(nodes[i].x, nodes[i].y);
                            ctx.lineTo(nodes[j].x, nodes[j].y);
                            if(nodes[i].active && nodes[j].active) {
                                ctx.strokeStyle = "rgba(239, 68, 68, 0.6)"; 
                            } else {
                                ctx.strokeStyle = "rgba(255, 255, 255, 0.05)"; 
                            }
                            ctx.stroke();
                        }
                    }
                }
                requestAnimationFrame(animate);
            }
            animate();
        }

        // --- RESEARCH VISUALIZATION 2: NEON WAVES ---
        function initOscillationViz() {
            const canvas = document.getElementById('oscillationCanvas');
            if(!canvas) return;
            const ctx = canvas.getContext('2d');
            
            function resize() {
                if(canvas.parentNode) {
                    canvas.width = canvas.parentNode.getBoundingClientRect().width;
                    canvas.height = canvas.parentNode.getBoundingClientRect().height;
                }
            }
            window.addEventListener('resize', resize);
            setTimeout(resize, 100);

            let time = 0;
            function animate() {
                if(!canvas.isConnected) return;
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                ctx.shadowBlur = 10;
                ctx.shadowColor = "#38bdf8";

                for(let i=0; i<3; i++) {
                    ctx.beginPath();
                    for(let x=0; x<canvas.width; x+=5) {
                        let y = canvas.height/2 + 
                                Math.sin(x*0.01 + time + i)*40 + 
                                Math.sin(x*0.03 - time)*10;
                        if(x===0) ctx.moveTo(x, y);
                        else ctx.lineTo(x, y);
                    }
                    ctx.strokeStyle = `rgba(56, 189, 248, ${0.8 - i*0.2})`; 
                    ctx.lineWidth = 2;
                    ctx.stroke();
                }
                time += 0.02;
                requestAnimationFrame(animate);
            }
            animate();
        }

        // Start Everything
        window.onload = function() {
            initBgCanvas();
            initNetworkViz();
            initOscillationViz();
        };
